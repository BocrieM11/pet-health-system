```javascript
// backend/services/cacheService.js
const redis = require('redis');

class CacheService {
  constructor() {
    this.client = redis.createClient({
      host: 'localhost',
      port: 6379
    });

    this.client.on('error', (err) => {
      console.error('Redis连接错误:', err);
    });

    this.client.connect();
  }

  // 获取缓存
  async get(key) {
    try {
      const data = await this.client.get(key);
      return data ? JSON.parse(data) : null;
    } catch (error) {
      console.error('缓存读取失败:', error);
      return null;
    }
  }

  // 设置缓存
  async set(key, value, expireSeconds = 600) {
    try {
      await this.client.setEx(key, expireSeconds, JSON.stringify(value));
    } catch (error) {
      console.error('缓存写入失败:', error);
    }
  }

  // 删除缓存
  async del(key) {
    try {
      await this.client.del(key);
    } catch (error) {
      console.error('缓存删除失败:', error);
    }
  }

  // 批量删除（使用模式匹配）
  async delPattern(pattern) {
    try {
      const keys = await this.client.keys(pattern);
      if (keys.length > 0) {
        await this.client.del(keys);
      }
    } catch (error) {
      console.error('缓存批量删除失败:', error);
    }
  }
}

module.exports = new CacheService();
```

```javascript
// backend/routes/pets.js
const express = require('express');
const router = express.Router();
const db = require('../database');
const { authenticateToken } = require('../middleware/auth');
const cache = require('../services/cacheService');

// 获取宠物列表（带缓存）
router.get('/', authenticateToken, async (req, res) => {
  const userId = req.user.userId;
  const cacheKey = `pets:user:${userId}`;

  try {
    // 1. 尝试从缓存获取
    const cachedData = await cache.get(cacheKey);
    if (cachedData) {
      console.log('✅ 缓存命中');
      return res.json(cachedData);
    }

    // 2. 缓存未命中，查询数据库
    db.all(
      'SELECT * FROM pets WHERE user_id = ? ORDER BY created_at DESC',
      [userId],
      async (err, pets) => {
        if (err) {
          return res.status(500).json({ error: '查询失败' });
        }

        // 3. 写入缓存（10分钟过期）
        await cache.set(cacheKey, pets, 600);

        res.json(pets);
      }
    );
  } catch (error) {
    console.error(error);
    res.status(500).json({ error: '服务器错误' });
  }
});

// 添加宠物（同时清除缓存）
router.post('/', authenticateToken, (req, res) => {
  const userId = req.user.userId;
  const { name, species, breed, gender, birth_date, weight, avatar } = req.body;

  db.run(
    `INSERT INTO pets (user_id, name, species, breed, gender, birth_date, weight, avatar) 
     VALUES (?, ?, ?, ?, ?, ?, ?, ?)`,
    [userId, name, species, breed, gender, birth_date, weight, avatar],
    async function(err) {
      if (err) {
        return res.status(500).json({ error: '添加失败' });
      }

      // 清除该用户的宠物列表缓存
      await cache.del(`pets:user:${userId}`);

      res.status(201).json({ 
        message: '添加成功', 
        petId: this.lastID 
      });
    }
  );
});

module.exports = router;
```

```javascript
const rateLimit = require('express-rate-limit');

// 通用限流：每IP每分钟最多100个请求
const generalLimiter = rateLimit({
  windowMs: 60 * 1000, // 1分钟
  max: 100,
  message: '请求过于频繁，请稍后再试',
  standardHeaders: true,
  legacyHeaders: false,
});

// 登录限流：每IP每15分钟最多5次登录尝试
const loginLimiter = rateLimit({
  windowMs: 15 * 60 * 1000,
  max: 5,
  message: '登录尝试次数过多，请15分钟后再试',
  skipSuccessfulRequests: true, // 成功的请求不计数
});

// AI分析限流：每用户每小时最多10次
const aiLimiter = rateLimit({
  windowMs: 60 * 60 * 1000,
  max: 10,
  message: 'AI分析次数已达上限，请1小时后再试',
  keyGenerator: (req) => req.user.userId, // 按用户限流
});

module.exports = {
  generalLimiter,
  loginLimiter,
  aiLimiter
};
```

```javascript
// backend/middleware/petPermission.js
const db = require('../database');
const { PERMISSIONS } = require('../constants/permissions');

// 检查用户对宠物的权限
const checkPetPermission = (requiredPermission) => {
  return async (req, res, next) => {
    const userId = req.user.userId;
    const petId = req.params.petId || req.params.id;

    try {
      // 1. 检查是否是宠物主人
      const pet = await new Promise((resolve, reject) => {
        db.get(
          'SELECT * FROM pets WHERE id = ?',
          [petId],
          (err, row) => {
            if (err) reject(err);
            else resolve(row);
          }
        );
      });

      if (!pet) {
        return res.status(404).json({ error: '宠物不存在' });
      }

      // 是主人，拥有所有权限
      if (pet.user_id === userId) {
        req.isOwner = true;
        req.permission = PERMISSIONS.ADMIN;
        return next();
      }

      // 2. 检查是否被分享
      const share = await new Promise((resolve, reject) => {
        db.get(
          'SELECT * FROM pet_shares WHERE pet_id = ? AND shared_with_user_id = ?',
          [petId, userId],
          (err, row) => {
            if (err) reject(err);
            else resolve(row);
          }
        );
      });

      if (!share) {
        return res.status(403).json({ error: '无权访问此宠物' });
      }

      // 3. 检查权限等级
      const permissionLevel = {
        'read': 1,
        'write': 2,
        'admin': 3
      };

      if (permissionLevel[share.permission] < permissionLevel[requiredPermission]) {
        return res.status(403).json({ error: '权限不足' });
      }

      req.isOwner = false;
      req.permission = share.permission;
      next();

    } catch (error) {
      console.error(error);
      res.status(500).json({ error: '权限检查失败' });
    }
  };
};

module.exports = { checkPetPermission };
```

```javascript
    const pet = await new Promise((resolve, reject) => {

          if (err) reject(err);
          else resolve(row);
        }
      );
    });

    if (!pet) {
      return res.status(403).json({ error: '只有宠物主人可以分享' });
    }

    // 2. 查找被分享的用户
    const targetUser = await new Promise((resolve, reject) => {
      db.get('SELECT id FROM users WHERE email = ?', 
        [shareWithEmail], 
        (err, row) => {
          if (err) reject(err);
          else resolve(row);
        }
      );
    });

    if (!targetUser) {
      return res.status(404).json({ error: '用户不存在' });
    }

    if (targetUser.id === ownerId) {
      return res.status(400).json({ error: '不能分享给自己' });
    }

    // 3. 检查是否已分享
    const existing = await new Promise((resolve, reject) => {
      db.get(
        'SELECT * FROM pet_shares WHERE pet_id = ? AND shared_with_user_id = ?',
        [petId, targetUser.id],
        (err, row) => {
          if (err) reject(err);
          else resolve(row);
        }
      );
    });

    if (existing) {
      // 更新权限
      db.run(
        'UPDATE pet_shares SET permission = ? WHERE id = ?',
        [permission, existing.id],
        (err) => {
          if (err) {
            return res.status(500).json({ error: '更新失败' });
          }
          res.json({ message: '权限已更新' });
        }
      );
    } else {
      // 新增分享

        'INSERT INTO pet_shares (pet_id, owner_id, shared_with_user_id, permission) VALUES (?, ?, ?, ?)',

        function(err) {
          if (err) {
            return res.status(500).json({ error: '分享失败' });
          }
          res.status(201).json({ 
            message: '分享成功',
            shareId: this.lastID 
          });
        }
      );
    }

  } catch (error) {
    console.error(error);
    res.status(500).json({ error: '服务器错误' });
  }
});
```

```javascript
// backend/middleware/logger.js
const db = require('../database');

const logActivity = (action, entityType) => {
  return (req, res, next) => {
    const originalSend = res.json;

    res.json = function(data) {
      // 仅在成功时记录（状态码2xx）
      if (res.statusCode >= 200 && res.statusCode < 300) {
        const log = {
          userId: req.user?.userId,
          petId: req.params.petId || req.params.id,
          action,
          entityType,
          entityId: data.id || data.petId,
          changes: JSON.stringify(req.body),
          ipAddress: req.ip
        };

        db.run(
          `INSERT INTO activity_logs (user_id, pet_id, action, entity_type, entity_id, changes, ip_address)
           VALUES (?, ?, ?, ?, ?, ?, ?)`,
          [log.userId, log.petId, log.action, log.entityType, log.entityId, log.changes, log.ipAddress],
          (err) => {
            if (err) console.error('日志记录失败:', err);
          }
        );
      }

      originalSend.call(this, data);
    };

    next();
  };
};

module.exports = { logActivity };
```

```javascript
// backend/server.js
const http = require('http');
const { Server } = require('socket.io');

const server = http.createServer(app);
const io = new Server(server, {
  cors: {
    origin: "http://localhost:3000",
    methods: ["GET", "POST"]
  }
});

// WebSocket连接处理
io.on('connection', (socket) => {
  console.log('用户连接:', socket.id);

  // 加入宠物房间（关注某只宠物的所有用户在同一个房间）
  socket.on('join-pet-room', (petId) => {
    socket.join(`pet:${petId}`);
    console.log(`用户加入宠物${petId}的房间`);
  });

  // 离开房间
  socket.on('leave-pet-room', (petId) => {
    socket.leave(`pet:${petId}`);
  });

  socket.on('disconnect', () => {
    console.log('用户断开:', socket.id);
  });
});

// 在路由中发送实时更新
// 例如：添加健康记录后
db.run('INSERT INTO health_records ...', [...], function(err) {
  if (!err) {
    // 通知该宠物房间的所有用户
    io.to(`pet:${petId}`).emit('health-record-added', {
      recordId: this.lastID,
      petId,
      addedBy: req.user.username
    });
  }
});

server.listen(3001, () => {
  console.log('服务器启动（支持WebSocket）');
});
```
